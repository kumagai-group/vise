60 @classmethod 

61 def from_options(cls, 

62 task: Task,

78 Returns: 

79 TaskIncarSettings instance object.

81 band_gap = vbm_cbm[1] - vbm_cbm[0] if vbm_cbm else None 

82 if band_gap: 

83 logger.info(f"Band gap: {round(band_gap, 3)} eV.") 

84 is_band_gap = band_gap > BAND_GAP_CRITERION 

85 else: 

86 is_band_gap = None 

87 

88 settings = \ 

89 load_default_incar_settings(yaml_filename="task_incar_set.yaml", 

90 required_flags=TASK_REQUIRED_FLAGS, 

91 optional_flags=TASK_OPTIONAL_FLAGS, 

92 key_name=str(task)) 

93 

94 settings["LREAL"] = "A" if task == Task.defect else False 

95 

96 # -5 is acceptable for num_kpt >= 4 and is_band_gap = True 

97 if is_band_gap is not True or num_kpoints < 4: 

98 settings["ISMEAR"] = 0 

99 

100 if task == Task.defect or is_magnetization: 

101 settings["ISPIN"] = 2 

102 if hasattr(structure, "site_properties"): 

103 if "magmom" in structure.site_properties: 

104 settings["MAGMOM"] = structure.site_properties["magmom"] 

105 

106 if npar_kpar: 

107 kpar, npar = calc_npar_kpar(num_kpoints, num_nodes) 

108 settings["KPAR"] = kpar 

109 # now switch off NPAR 

110# settings["NPAR"] = npar 

111 

112 if not encut: 

113 if task in LATTICE_RELAX_TASK: 

114 encut = round(max_enmax * structure_opt_encut_factor, 3) 

115 else: 

116 encut = max_enmax 

117 settings["ENCUT"] = encut 

118 

119 if task in PLOT_TASK: 

120 settings["NBANDS"] = nbands(structure.composition, potcar) 

121 

122 if task in SPECTRA_TASK: 

123 if vbm_cbm: 

124 emin = ceil(vbm_cbm[0]) - 15 - dos_step_size 

125 emax = ceil(vbm_cbm[1]) + 15 

126 else: 

127 emin = -20 - dos_step_size 

128 emax = 20 

129 nedos = int(round((emax - emin) / dos_step_size, 0)) + 1 

130 settings.update({"EMIN": emin, "EMAX": emax, "NEDOS": nedos}) 

131 

132 if task == Task.dielectric_dfpt: 

133 settings["LEPSILON"] = True 

134 elif task == Task.dielectric_finite_field: 

135 settings["LCALCEPS"] = True 

136 settings["IBRION"] = 6 

137 settings["POTIM"] = 0.015 

138 elif task == Task.dielectric_function: 

139 settings["LOPTICS"] = True 

140 settings["CSHIFT"] = 0.01 

141 

142 return cls(settings=settings)

152 def from_options(cls, 

153 xc: Xc, 

154 symbol_list: list, 

155 factor: int, 

156 aexx: Optional[float] = 0.25, 

157 hubbard_u: Optional[bool] = None, 

158 ldauu: Optional[dict] = None, 

159 ldaul: Optional[dict] = None, 

160 ldaul_set_name: Optional[str] = "default" 

161 ) -> "XcIncarSettings": 

162 """Construct incar settings related to xc with some options. 

163 

164 Args: See ViseInputSet docstrings 

165 

166 Returns: 

167 XcIncarSettings instance object. 

168 """ 

169 settings = \ 

170 load_default_incar_settings(yaml_filename="xc_incar_set.yaml", 

171 required_flags=XC_REQUIRED_FLAGS, 

172 optional_flags=XC_OPTIONAL_FLAGS, 

173 key_name=str(xc)) 

174 

175 if ldauu: 

176 hubbard_u = True 

177 # By default Hubbard U is set for LDA or GGA. 

178 hubbard_u = xc in LDA_OR_GGA if hubbard_u is None else hubbard_u 

179 ldauu = ldauu or {} 

180 ldaul = ldaul or {} 

181 

182 if xc == Xc.pbesol: 

183 settings["GGA"] = "PS" 

184 elif xc == Xc.scan: 

185 settings["METAGGA"] = "SCAN" 

186 

187 if hubbard_u: 

188 u_set = loadfn(SET_DIR / "u_parameter_set.yaml") 

189 ldauu_set = u_set["LDAUU"][ldaul_set_name] 

190 ldauu_set.update(ldauu) 

191 ldauu = [ldauu_set.get(el, 0) for el in symbol_list] 

192 

193 if sum(ldauu) > 0: 

194 settings["LDAUU"] = ldauu 

195 settings.update({"LDAU": True, "LDAUTYPE": 2, "LDAUPRINT": 1}) 

196 ldaul_set = u_set["LDAUL"][ldaul_set_name] 

197 ldaul_set.update(ldaul) 

198 settings["LDAUL"] = \ 

199 [ldaul_set.get(el, -1) for el in symbol_list] 

200 settings["LMAXMIX"] = \ 

201 6 if any([Element(el).Z > 56 for el in symbol_list]) else 4 

202 

203 if xc in HYBRID_FUNCTIONAL: 

204 settings["AEXX"] = aexx 

205 if factor > 1: 

206 settings["NKRED"] = factor 

207 

208 return cls(settings=settings)
