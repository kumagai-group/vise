16def potcar_str2dict(potcar_list: Union[str, List[str], None]) -> dict: 

17 """Sanitize potcar names to dict. 

18 

19 If potcar_list is None, {} is returned. 

20 Examples 

21 ["Mg_pv", O_h"] -> {"Mg": "Mg_pv", "O": "O_h"} 

22 "Mg_pv" -> {"Mg": "Mg_pv"} 

23 None -> {} 

24 

25 Args: 

26 potcar_list (str/list/None): 

27 List of Potcar names or a name, including element names before the 

28 hyphen. 

29 

30 Returns: 

31 Dictionary with element names as keys and potcar names as values. 

32 """ 

33 if potcar_list is None: 

34 return {} 

35 elif isinstance(potcar_list, str): 

36 potcar_list = [potcar_list] 

37 

38 d = {} 

39 for p in potcar_list: 

40 element = p.split("_")[0] 

41 if element in d: 

42 raise ValueError("Multiple POTCAR files for an element are not " 

43 "supported yet.") 

44 try: 

45 Element(element) 

46 except ValueError: 

47 raise ValueError(f"First part of POTCAR file name {p} is not an " 

48 f"element supported yet.") 

49 

50 d[element] = p 

51 return d

54def list2dict(flattened_list: Optional[list], key_candidates: list) -> dict: 

55 """Sanitize a list to a dict with keys in the flags 

56 

57 If a string in list does not exist in key_candidates, raise ValueError. 

58 

59 Examples: 

60 flattened_list = ["ENCUT", "500", "MAGMOM", "4", "4", "LWAVE", "F"] 

61 key_candidates = ["ENCUT", "MAGMOM", "LWAVE", ...] 

62 list2dict(flattened_list, key_candidates) = 

63 {"ENCUT": 500, "MAGMOM": [4, 4], "LWAVE": False} 

64 

65 flattened_list = ["ENCUT", "500", "MAGMAM", "4", "4"] 

66 list2dict(flattened_list, key_candidates) raises ValueError 

67 

68 Args: 

69 flattened_list (list): 

70 Input list. 

71 key_candidates (list): 

72 List of key candidates, e.g., INCAR flags. 

73 

74 Returns: 

75 Sanitized dict 

76 """ 

77 flattened_list = flattened_list or [] 

78 

79 d = {} 

80 key = None # key is None at the beginning or after inserting a value. 

81 value_list = [] 

82 

83 def insert(): 

84 if not value_list: 

85 raise ValueError(f"Invalid input: {flattened_list}.") 

86 d[key] = value_list[0] if len(value_list) == 1 else value_list 

87 

88 for string in flattened_list: 

89 if key is None and string not in key_candidates: 

90 raise ValueError(f"Keys are invalid: {flattened_list}.") 

91 elif string in key_candidates: 

92 if key: 

93 insert() 

94 key = None 

95 value_list = [] 

96 key = string 

97 else: 

98 try: 

99 value = strtobool(string) 

100 except ValueError: 

101 if is_str_int(string): 

102 value = int(string) 

103 elif is_str_digit(string): 

104 value = float(string) 

105 else: 

106 value = string 

107 

108 value_list.append(value) 

109 else: 

110 if key: 

111 insert() 

112 

113 return d

116def dict2list(d: dict) -> list: 

117 """Flatten dict to list w/ string keys. String is also separated by space. 

118 

119 Example: 

120 dict2list({"a": 1, "b": "2 3 4", "c": True}) = 

121 ["a", "1", "b", "2", "3", "4", "c", "True"] 

122 

123 Args: 

124 d (dict): 

125 Input dict. 

126 

127 Returns: 

128 List of flattened dict. 

129 """ 

130 

131 d = d or {} 

132 flattened_list = [] 

133 for k, v in d.items(): 

134 flattened_list.append(k) 

135 if isinstance(v, str): 

136 flattened_list.extend(v.split()) 

137 else: 

138 flattened_list.append(str(v)) 

139 

140 return flattened_list

143def get_user_settings(yaml_filename: str, 

144 setting_keys: list) -> Tuple[dict, Optional[Path]]: 

145 """Get the user specifying settings written in yaml_filename 

146 

147 Note1: The yaml_filename is explored in the parent folders up to home 

148 or root directory until it's found. If it does not exist, empty 

149 dictionary is returned. 

150 Note2: When the key includes "/", the absolute path is added as a prefix. 

151 E.g., unitcell/unitcell.json -> /something/../unitcell/unitcell.json 

152 Note3: The value of "potcar_set: Mg_pv O_h" is string of "Mg_pv O_h", which 

153 is suited for main default value. 

154 

155 Args: 

156 yaml_filename (str): 

157 User setting yaml filename. 

158 setting_keys (list): 

159 Only setting_keys are valid as input keys, otherwise raise 

160 ValueError. 

161 

162 Returns: 

163 Tuple of dictionary of configs and Path to the config file. 

164 """ 

165 

166 config_path = Path.cwd() 

167 home = Path.home() 

168 

169 while True: 

170 if config_path == home or config_path == Path("/"): 

171 return {}, None 

172 

173 f = config_path / yaml_filename 

174 if f.exists(): 

175 with open(f, "r") as fin: 

176 user_settings = yaml.load(fin, Loader=yaml.FullLoader) 

177 break 

178 

179 else: 

180 config_path = config_path.parent 

181 

182 user_settings = user_settings or {} 

183 

184 # Add full path 

185 for k, v in user_settings.items(): 

186 if k not in setting_keys: 

187 raise ValueError(f"Key {k} in {yaml_filename} is invalid." 

188 f"The candidate keys are {setting_keys}") 

189 if isinstance(v, str) and re.match(r'\S*/\S*', v): 

190 user_settings[k] = str(config_path / v) 

191 

192 return user_settings, f

195def get_default_args(function: Callable) -> dict: 

196 """Get the default values of the arguments in the method/function. 

197 

198 inspect._empty means no default. 

199 

200 Args: 

201 function (Callable): 

202 Method or function. when class is inserted, cls.__init__ is called. 

203 

204 Returns: 

205 Dict of default values. 

206 """ 

207 defaults = {} 

208 signature_obj = signature(function) 

209 for name, param in signature_obj.parameters.items(): 

210 if param.default != _empty: 

211 defaults[name] = param.default 

212 

213 return defaults
