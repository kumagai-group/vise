78def check_keys(d: dict, required: set, optional: set) -> bool: 

79 """Check if the required keys exist and other keys are included in optional 

80 

81 Args: 

82 d (dict): 

83 Dictionary to be checked. 

84 required (set): 

85 Required key set. 

86 optional (set): 

87 Optional key set. 

88 

89 Returns: 

90 Simply raise KeyError if the condition is not satisfied. 

91 If succeed, True is returned. 

92 """ 

93 check_required = set(required) 

94 for key in d: 

95 check_required.discard(key) 

96 if key not in required | optional: 

97 raise KeyError(f"{key} does not belong.") 

98 

99 if check_required: 

100 raise KeyError(f"{check_required} must exist in {d}.") 

101 

102 return True

105def nelect(composition: Composition, potcar: Potcar, charge: int = 0) -> int: 

106 """Gets the default number of electrons for a given composition. """ 

107 num_nelect = 0 

108 for pt in potcar: 

109 num_nelect += composition.element_composition[pt.element] * pt.ZVAL 

110 

111 return int(num_nelect) - charge

114def nbands(composition: Composition, potcar: Potcar) -> int: 

115 """Calculate total number of bands for unoccupied related properties 

116 

117 Useful for optical absorption, band structure, and DOS. 

118 

119 Args: 

120 composition (Composition): Input composition 

121 potcar (Potcar): Input Potcar object 

122 

123 Returns: 

124 Int of number of bands 

125 """ 

126 num_bands = \ 

127 sum([composition[c] * (p.nelectrons / 2 + unoccupied_bands[str(c)]) 

128 for c, p in zip(composition, potcar)]) 

129 

130 return ceil(num_bands)
